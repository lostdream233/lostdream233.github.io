从零造轮子——低能CPU

计组 verilog
发布日期:   2022-11-19
阅读次数:  

写在前面

不用+、-、*、>、<，从零造一个简易运算的CPU，嵌入式——不入门直接劝退(。

一、CPU功能及基本构成

说明

使用八位二进制串（ b7b6b5b4b3b2b1b0）表示指令，与开发板上的八个开关对应。
实验使用四位二进制补码作为输入，与开发板上的四个开关对应。 三个寄存器 R0，R1，R2 分别对应二进制地址码 00、01、10。

1、运算指令格式

高四位(b7b6b5b4)为操作码，次低两位（b3b2）为地址码（指明存放第一个操作数的寄存器），最低两位(b1b0)为地址码（指明存放第二个操作数的寄存器）。
有符号乘法的结果若超过 8 比特解释为溢出。
0000 -------> A & B
0001 -------> A | B
0010 -------> A + B
0110 -------> A - B
0111 -------> 判断 （A < B）
1100 -------> ~（A | B）
1101 -------> ~（A & B）
1110 -------> A * B

2、存储指令格式

最高两位（b7b6）为操作码置 10，次高两位（b5b4）为写入寄存器的地址码，低四位（b3b2b1b0）为待写入数据的二进制补码。

3、读取指令格式

最高六位（b7b6b5b4b3b2）为操作码置 111100，最低两位（b1b0）为地址码。要求该指令可读取寄存器的值并以十进制通过数码管显示。

二、子模块

1、加法器（1位、8位）

（1） 1位加法器

module adder(
    input a,  		// 1位二进制加数
    input b,      // 1位二进制加数
    input cin, 		// 低位来的进位信号
    output cout, 	// 向高位的进位信号
    output sum 		// 1位和数
    );
    
    assign cout = (a&b) | (a&cin) | (b&cin);  	// 当三个输入其中两个为1时，产生进位
    assign sum = a^b^cin;                       // 和数等于三个输入异或的值
    
endmodule

（2） 8位加法器（减法器）

module adder_8(
    input [7:0] a,      //八位有符号数补码
    input [7:0] b,      //八位有符号数补码
    input cin,          //低位来的进位信号
    input operator,     //决定加法还是减法。 0表示加法，1表示减法
    output [7:0] res
    );
    
    wire cin_pre = cin ^ operator;
    wire [7:0] b_pre =  b ^ {8{operator}} ;
    
    //八位加法
    wire c0,c1,c2,c3,c4,c5,c6,cf;
    adder U0(a[0],b_pre[0],cin_pre,c0,res[0]);
    adder U1(a[1],b_pre[1],c0,c1,res[1]);
    adder U2(a[2],b_pre[2],c1,c2,res[2]);
    adder U3(a[3],b_pre[3],c2,c3,res[3]);
    adder U4(a[4],b_pre[4],c3,c4,res[4]);
    adder U5(a[5],b_pre[5],c4,c5,res[5]);
    adder U6(a[6],b_pre[6],c5,c6,res[6]);
    adder U7(a[7],b_pre[7],c6,cf,res[7]);
    
endmodule

2、8位乘法器

由于不能使用+，只能采取最简朴的移位加法进行乘法计算。
穷举弱智，但有用。

module mul_8(
    input [7:0] a, 
    input [7:0] b, 
    output [7:0] res // 乘积输出 res
    );
    
    //判断a,b的符号位
    wire a_sign = a[7];
    wire b_sign = b[7];
    //a,b符号位不同则结果取反+1
    wire res_sign = a_sign^b_sign;
    //计算a,b的取反+1
    wire [7:0] a_x, b_x;
    adder_8 X1(~a, 1, 1'b0, 1'b0, a_x);
    adder_8 X2(~b, 1, 1'b0, 1'b0, b_x);
    //为负数则取反+1
    wire [7:0] a_pre = a_sign ? a_x : a;
    wire [7:0] b_pre = b_sign ? b_x : b;
    
    //计算被乘数所有移位的结果
    wire [7:0] B0,B1,B2,B3,B4,B5,B6,B7;
    assign B0 = b_pre[0] ? a_pre : 0;
    assign B1 = b_pre[1] ? a_pre<<1 : 0;
    assign B2 = b_pre[2] ? a_pre<<2 : 0;
    assign B3 = b_pre[3] ? a_pre<<3 : 0;
    assign B4 = b_pre[4] ? a_pre<<4 : 0;
    assign B5 = b_pre[5] ? a_pre<<5 : 0;
    assign B6 = b_pre[6] ? a_pre<<6 : 0;
    assign B7 = b_pre[7] ? a_pre<<7 : 0;
    
    //将所有移位结果累加
    wire [7:0] T0,T1,T2,T3,T4,T5,T6,T7;
    adder_8 U0(0, B0, 1'b0, 1'b0, T0);
    adder_8 U1(T0, B1, 1'b0, 1'b0, T1);
    adder_8 U2(T1, B2, 1'b0, 1'b0, T2);
    adder_8 U3(T2, B3, 1'b0, 1'b0, T3);
    adder_8 U4(T3, B4, 1'b0, 1'b0, T4);
    adder_8 U5(T4, B5, 1'b0, 1'b0, T5);
    adder_8 U6(T5, B6, 1'b0, 1'b0, T6);
    adder_8 U7(T6, B7, 1'b0, 1'b0, T7);
	
    //判断结果的符号
    wire [7:0] T7_x;
    adder_8 X3(~T7, 1, 1'b0, 1'b0, T7_x);
    assign res = res_sign ? T7_x : T7;
   
endmodule

3、SLT模块

先计算两数相减的结果，然后根据结果的符号位判断大小。

module slt(
    input [7:0] a,
    input [7:0] b,
    output jud
    );
    wire [7:0] res_sub;
    adder_8(a, b, 1'b0, 1'b1, res_sub);
    
    assign jud = res_sub[7] == 1'b0 ? 1'b0 : 1'b1;
    
endmodule

4、十进制信号灯

忙死了懒得看了，回头再补(。

module bin2dec (
    input clk,          //时钟
    input [7:0] n,
    output reg [7:0] led_id,
    output reg [6:0] out_led
    );

parameter CLK_COUNT = 249999;//时钟计数上限
reg [31:0] count;//计数
reg [1:0] id;//id0~3对应左到右四个数码管
wire flag;  //flag标记补码是否表示负数
assign flag = n[7];

//8位2进制，十进制至多3位
reg [7:0] n1;    //百位
reg [7:0] n2;    //十位
reg [7:0] n3;    //个位
reg [7:0] abs;  

always @(*) begin
    case(flag) //求正数的个十百位
    1'b0:
		begin
		   n1 = n / 100 % 10;
		   n2 = n / 10 % 10;
		   n3 = n % 10;
		end
    1'b1:
		begin
		   abs=(~n)+1;  //求负数的个十百位
		   n1 = abs / 100 % 10;        
		   n2 = abs / 10 % 10;        
		   n3 = abs % 10;
		end
    endcase
end

always @ (posedge clk) begin //时钟上升沿 
//根据时钟信号控制切换显示的数码管
    if (count == CLK_COUNT) begin
       count <= 0;
       id <= (id + 1); //切换
    end
    else count <= count+1;
end
   
   
   //选择灯，显示数字
always @(id) begin        
    if (id == 0) begin           
        led_id = 8'b1111_0111;//最左端灯亮
        if(flag) out_led = 7'b1111110;//负号
        else out_led = 7'b1111111;//不显示
    end
    else if (id == 1) begin
        led_id = 8'b1111_1011;
        case(n1)
            4'b0000: out_led = 7'b0000001;    //0
            4'b0001: out_led = 7'b1001111;    //1
            4'b0010: out_led = 7'b0010010;    //2
            4'b0011: out_led = 7'b0000110;    //3
            4'b0100: out_led = 7'b1001100;    //4
            4'b0101: out_led = 7'b0100100;    //5
            4'b0110: out_led = 7'b0100000;    //6
            4'b0111: out_led = 7'b0001111;    //7
            4'b1000: out_led = 7'b0000000;    //8
            4'b1001: out_led = 7'b0000100;    //9
        endcase
    end
    else if (id == 2) begin
        led_id = 8'b1111_1101;
        case(n2)
            4'b0000: out_led = 7'b0000001;    //0
            4'b0001: out_led = 7'b1001111;    //1
            4'b0010: out_led = 7'b0010010;    //2
            4'b0011: out_led = 7'b0000110;    //3
            4'b0100: out_led = 7'b1001100;    //4
            4'b0101: out_led = 7'b0100100;    //5
            4'b0110: out_led = 7'b0100000;    //6
            4'b0111: out_led = 7'b0001111;    //7
            4'b1000: out_led = 7'b0000000;    //8
            4'b1001: out_led = 7'b0000100;    //9
         endcase
    end
    else if (id == 3) begin
        led_id = 8'b1111_1110;
		case(n3)
			4'b0000: out_led = 7'b0000001;    //0
			4'b0001: out_led = 7'b1001111;    //1
			4'b0010: out_led = 7'b0010010;    //2
			4'b0011: out_led = 7'b0000110;    //3
			4'b0100: out_led = 7'b1001100;    //4
			4'b0101: out_led = 7'b0100100;    //5
			4'b0110: out_led = 7'b0100000;    //6
			4'b0111: out_led = 7'b0001111;    //7
			4'b1000: out_led = 7'b0000000;    //8
			4'b1001: out_led = 7'b0000100;    //9
		endcase
    end 
end

endmodule

三、CPU主模块

module CPU(
    input store,            //判断是否将结果存储到R2
    input clk,          		//时钟信号
    input rst,          		//置位信号
    input en,           		//开关
    input [7:0] op,     		//操作码   
    output [7:0] led_id,
    output [6:0] out_led
    );
    wire [7:0] res_clk;
    reg [7:0] res;
    reg [7:0] MEM [2:0];    	//R0 R1 R2
    reg [1:0] addr;         	//寄存器寻址
    wire [7:0] A, B;         	//ALU运算中的两个操作数
    wire [7:0] res_add,res_sub,res_mul,res_slt;
    //取两个操作数
    assign A = MEM[op[3:2]];
    assign B = MEM[op[1:0]];
    //不能在always语句内实例化
    //提前计算加法、减法、乘法、比较的结果
    adder_8 P0(A, B, 1'b0, 1'b0, res_add);
    adder_8 P1(A, B, 1'b0, 1'b1, res_sub);
    mul_8   P2(A, B, res_mul);
    slt     P3(A, B, res_slt);

    always@(op) begin
        //置位信号 低电平清零
        if(!rst) begin
            MEM[0] = 0;
            MEM[1] = 0;
            MEM[2] = 0;
            res = 0;
        end  
        //开关高电平 进行操作
        else if(en) begin
			//决定是否存入R2
            if(store == 1'b1) MEM[2] = res;
            //写
            else if(op[7:6] == 2'b10) begin
                //寻址
                addr = op[5:4];
                //符号位拓展
                MEM[addr][3:0] = op[3:0];
                MEM[addr][7:4] = op[3] ? 4'b1111 : 4'b0000;
            end
            //读
            else if(op[7:2] == 6'b111100) begin
                addr = op[1:0];
                res = MEM[addr];
            end
            //ALU运算
            else if(op[7:6] == 2'b00 || op[7:6] == 2'b11 || op[7:6] == 2'b10 || op[7:6] == 2'b01) begin
                //case判断操作码
                case(op[7:4])
                    4'b0000: res = A & B;
                    4'b0001: res = A | B;
                    4'b0010: res = res_add;
                    4'b0110: res = res_sub;
                    4'b0111: res = res_slt;
                    4'b1100: res = ~(A | B);
                    4'b1101: res = ~(A & B);
                    4'b1110: res = res_mul;
                endcase     
            end     
        end
        
    end
    
    //根据store信号决定输出
    assign res_clk = store ? MEM[2] : res;
    //实例化信号灯模块
    bin2dec LED(clk, res_clk, led_id, out_led);
    
endmodule
