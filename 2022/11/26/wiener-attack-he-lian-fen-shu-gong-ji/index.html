#?有理数转化为连分数形式
def To_ConFrac(a: int, b: int) -> list:
    arr = []
    while b != 0:
        arr.append(a // b)  #?取整数部分
        a, b = b, a % b  #?继续计算余下分数
    return arr


#?根据连分数列表化为分数形式
def Cal_ConFrac(FracList: list) -> list:
    #?分子和分母
    numerator, denominator = 0, 1
    #?对连分数列表中的数倒序求解
    for i in FracList[::-1]:
        numerator, denominator = i * numerator + denominator, numerator
    return numerator, denominator


#?根据Wiener算法计算u、v，从而解出p、q
def Cal_pq(n: int, Phi: int) -> list:
    u = (n + 1 - Phi) // 2
    #?math.sqrt算不了大整数,使用gmpy2库
    v = gmpy2.isqrt(pow(u, 2) - n)
    p, q = abs(u + v), abs(u - v)
    return p, q


def WienerAttack_ConFrac(e: int, n: int) -> list:
    FracList = To_ConFrac(e, n)
    #?遍历连分数列表，寻找收敛子
    for i in range(1, len(FracList) + 1):
        frac = Cal_ConFrac(FracList[:i])

        #?k/d的分数形式
        k = frac[0]  #?分子k
        d = frac[1]  #?分母d

        if k == 0:
            continue
        #?e*d = k*Phi + 1 ----> (e*d - 1) % k == 0
        if (e * d - 1) % k != 0:
            continue

        Phi = (e * d - 1) // k
        p, q = Cal_pq(n, Phi)

        if (p * q == n and p != 1 * q != 1):
            return p, q
    Exception('无法分解')
